class Solution(object):
    def hIndex(self, citations):
        """
        :type citations: List[int]
        :rtype: int
        """
        if len(citations)==0: return 0
        citations.sort()
        citations.reverse();
        for i in range(len(citations)):
            if citations[i]<i+1:
                return i
        return i+1
